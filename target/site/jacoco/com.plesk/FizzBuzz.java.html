package com.plesk;

class FizzBuzz {

    public String processNumber(int toConvertNo) {

        if ((toConvertNo % 15) == 0) {
            return "FizzBuzz";
        }
        else if ((toConvertNo % 3) == 0) {
            return "Fizz";
        }
        else if ((toConvertNo % 5) == 0) {
            return  "Buzz";
        }
        else {
            return String.valueOf(toConvertNo);
        }
    }
}
